mercedes_analysis.sql 

-- Calculate YoY Growth & Model Performance
WITH SalesTrends AS (
    SELECT
        model_name,
        fuel_type,
        SUM(sales_amount) AS revenue,
        DATE_TRUNC('month', sale_date) AS sale_month
    FROM mb_sales_data
    GROUP BY model_name, fuel_type, sale_month
)
SELECT 
    model_name,
    fuel_type,
    revenue,
    LAG(revenue) OVER (PARTITION BY model_name ORDER BY sale_month) AS prev_month_revenue
FROM SalesTrends
ORDER BY revenue DESC;

ecommerce_analysis.sql 

-- Calculate Customer Retention & Repeat Purchase Rate
WITH CustomerFirstPurchase AS (
    SELECT 
        customer_id, 
        MIN(order_date) AS first_order_date
    FROM orders
    GROUP BY customer_id
),
Cohorts AS (
    SELECT 
        c.customer_id,
        DATE_TRUNC('month', c.first_order_date) AS cohort_month,
        o.order_date
    FROM CustomerFirstPurchase c
    JOIN orders o ON c.customer_id = o.customer_id
)
SELECT 
    cohort_month,
    COUNT(DISTINCT customer_id) AS total_customers,
    SUM(CASE WHEN order_date > cohort_month THEN 1 ELSE 0 END) AS repeat_customers
FROM Cohorts
GROUP BY cohort_month
ORDER BY cohort_month DESC;
